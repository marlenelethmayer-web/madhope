<body>

    <div id="container"></div>


    <script>

        const videoIds = [

            "3X_BYb9ws9Q", // Zábavní bazén

            "VmGrd3VlA2Y", // Aquapark UH

            "pPJKse_k50s"  // Armdale Roundabout

        ];


        let currentIndex = 0;

        const displayTime = 2 * 60 * 1000; // 2 Minuten


        function loadNextStream() {

            const container = document.getElementById('container');

            const id = videoIds[currentIndex];

            

            // YouTube-Parameter für maximale "Sauberkeit":

            // modestbranding=1 & controls=0 versteckt fast alle UI-Elemente

            container.innerHTML = `

                <iframe src="https://www.youtube.com/embed/${id}?autoplay=1&mute=1&controls=0&showinfo=0&rel=0&iv_load_policy=3&modestbranding=1&disablekb=1" 

                allow="autoplay; encrypted-media; fullscreen"></iframe>

            `;


            currentIndex = (currentIndex + 1) % videoIds.length;

        }


        setInterval(loadNextStream, displayTime);

        loadNextStream();

    </script>
